** Setup Internet
ip a
/etc/netctl
netctl enable *profile*
netctl start *profile*

** Set language
vim /etc/locale.gen
locale-gen

** Set time zone
ln -sf /usr/share/timeinfo/US/Eastern /etc/localtime

** Change hostname
echo *hostname* > /etc/hostname
useradd -m -g users -G wheel -s /bin/bash *username*
passwd *username*
vim /etc/sudoers

** Update System
pacman -Syu

** Add base-devel
pacman -S base-devel
pacman -S vim
